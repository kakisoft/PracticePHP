<?php

namespace App\Repositories;

use App\Entities\Administrator;
use App\Http\Requests\Contracts\AdministratorSearchInterface;
use Illuminate\Support\Facades\Cache;

class AdministratorRepository implements Contracts\AdministratorRepositoryInterface
{
    /** @var Administrator */
    protected $eloquent;

    /** @var DataCacheInterface */
    protected $cache;

    /**
     * コンストラクタ
     *
     * @param Administrator $eloquent
     */
    public function __construct(Administrator $eloquent)
    {
        $this->eloquent = $eloquent;
    }

    /**
     * 保存
     *
     * @param  $id
     * @param  array $params
     * @return Administrator
     */
    public function save($id, array $params)
    {
        return $this->eloquent->updateOrCreate(['id' => $id], $params);
    }
